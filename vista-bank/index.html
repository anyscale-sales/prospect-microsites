# This code snippet demonstrates how Vista Bank can utilize Ray Data on Anyscale for distributed batch
# model training and scoring, focusing on fraud detection and risk management pipelines.

import ray
from ray.data.preprocessors import StandardScaler
from ray.data import read_csv
from ray.train.xgboost import XGBoostTrainer

# Initialize Ray
ray.init()

# Load and preprocess data
# Assuming 'transactions.csv' contains Vista Bank's transaction data for fraud detection
data = read_csv("/path/to/transactions.csv")
preprocessor = StandardScaler(columns=["amount", "transaction_count"])
data = preprocessor.fit_transform(data)

# Define and train the model
trainer = XGBoostTrainer(
    scaling_config={"num_workers": 4},
    label_column="fraud_label",
    params={"objective": "binary:logistic", "eval_metric": "logloss"},
    num_boost_round=100,
    datasets={"train": data}
)
model = trainer.fit()

# Save or deploy the model for scoring
model.save("/path/to/model")
