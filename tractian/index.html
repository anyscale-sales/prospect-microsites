# Tractian's use case: Distributed time-series feature engineering and batch inference
import ray
from ray import tune

# Define a simple objective function to optimize
def objective(config):
    # Simulate a training process
    accuracy = (config['x'] ** 2) - (config['y'] ** 2)
    tune.report(accuracy=accuracy)

# Configure Ray Tune's hyperparameter search
search_space = {
    'x': tune.uniform(-10, 10),
    'y': tune.uniform(-10, 10)
}

# Execute the hyperparameter optimization
analysis = tune.run(
    objective,
    config=search_space,
    num_samples=50
)

print("Best config: ", analysis.get_best_config(metric="accuracy", mode="max"))
