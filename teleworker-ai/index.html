import multiprocessing

import os
import subprocess
import time

# Function to process a single drawing
def process_drawing(drawing_id):
    try:
        # Simulate drawing processing
        result = subprocess.run(['process_drawing_tool', drawing_id], check=True)
        return result
    except Exception as e:
        print(f"Error processing {drawing_id}: {e}")
        return None

# Manual worker management
def worker_manager(drawing_ids):
    results = []
    with multiprocessing.Pool(processes=4) as pool:
        for drawing_id in drawing_ids:
            try:
                result = pool.apply_async(process_drawing, (drawing_id,))
                results.append(result.get())
            except Exception as e:
                print(f"Error in worker manager: {e}")
    return results

# Checkpoint results manually
def checkpoint_results(results):
    s3 = boto3.client('s3')
    for result in results:
        if result:
            try:
                s3.put_object(Bucket='teleworker-ai-results', Key=f'{result}.json', Body=str(result))
            except Exception as e:
                print(f"Failed to save {result} to S3: {e}")

# Example usage
if __name__ == '__main__':
    drawing_ids = ['drawing1', 'drawing2', 'drawing3']
    results = worker_manager(drawing_ids)
    checkpoint_results(results)
