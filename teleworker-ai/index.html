import multiprocessing
import boto3
import time
import os

# Manual job queue management
job_queue = multiprocessing.Queue()
results = []

# Manual worker spin-up using subprocess
for _ in range(4):
    worker = multiprocessing.Process(target=process_document, args=(job_queue,))
    worker.start()

# Manual error handling and retries
try:
    while not job_queue.empty():
        job = job_queue.get()
        try:
            result = process_document(job)
            results.append(result)
        except Exception as e:
            print(f"Error processing document: {e}")
            time.sleep(5)  # Retry after delay
            job_queue.put(job)  # Requeue the job
except KeyboardInterrupt:
    print("Interrupted, saving progress...")
    with open('checkpoint.txt', 'w') as f:
        f.write(str(results))
